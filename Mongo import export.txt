-----LOCAL EXPORT run this to export authors collection from classroom database
mongoexport --db classroom --collection users --out users.json
mongoexport --db classroom --collection subjects --out subjects.json
mongoexport --db classroom --collection lessons --out lessons.json

-----LOCAL IMPORT 
mongoimport --db classroom --collection subjects --type json subjects.json
mongoimport --db classroom --collection users --type json users.json
mongoimport --db classroom --collection lessons --type json lessons.json

------Binary import export

Import collection
mongorestore -h ds249565.mlab.com:49565 -d classroom-prod -c lessons -u <user> -p <password> lessons.bson

Export collection
mongodump -h ds249565.mlab.com:49565 -d classroom-prod -c lessons -u <user> -p <password> -o <output directory>

-------JSON import export

Import collection
mongoimport -h ds249565.mlab.com:49565 -d classroom-prod -c lessons -u <user> -p <password> --file <input file>

Export collection
mongoexport -h ds249565.mlab.com:49565 -d classroom-prod -c lessons -u <user> -p <password> -o lessons.json

--------CSV import export

Import collection
mongoimport -h ds249565.mlab.com:49565 -d classroom-prod -c lessons -u <user> -p <password> --file <input .csv file> --type csv --headerline

Export collection
mongoexport -h ds249565.mlab.com:49565 -d classroom-prod -c lessons -u <user> -p <password> -o lessons.csv --csv -f <comma-separated list of field names>
